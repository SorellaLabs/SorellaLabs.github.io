use malachite_base::num::basic::traits::{NegativeOne, One};
use malachite_nz::integer::Integer;

let mut x = Integer::ONE;
x <<= 1u8;
x <<= 2u16;
x <<= 3u32;
x <<= 4u64;
assert_eq!(x, 1024);
let mut x = Integer::NEGATIVE_ONE;
x <<= 1u8;
x <<= 2u16;
x <<= 3u32;
x <<= 4u64;
assert_eq!(x, -1024);

let mut x = Integer::ONE;
x <<= 1i8;
x <<= 2i16;
x <<= 3i32;
x <<= 4i64;
assert_eq!(x, 1024);
let mut x = Integer::NEGATIVE_ONE;
x <<= 1i8;
x <<= 2i16;
x <<= 3i32;
x <<= 4i64;
assert_eq!(x, -1024);

let mut x = Integer::from(1024);
x <<= -1i8;
x <<= -2i16;
x <<= -3i32;
x <<= -4i64;
assert_eq!(x, 1);
